    // ── Estado Global ──

    let state = {

      storeName: '',

      userName: '',

      mlConnected: false,

      mlToken: null,

      listings: [],

      stock: [],

      messages: [],

    // ── Inicialização ──

    document.addEventListener('DOMContentLoaded', () => {

      loadState();

      checkAuth();

      setupEventListeners();

    function loadState() {

      try {

        const saved = localStorage.getItem('sellhub_state');

        if (saved) {

          const parsed = JSON.parse(saved);

          state = { ...state, ...parsed };

        }

      } catch (e) {

        console.error('Erro ao carregar estado:', e);

      }

    }


    function saveState() {

      try {

        localStorage.setItem('sellhub_state', JSON.stringify(state));

      } catch (e) {

        console.error('Erro ao salvar estado:', e);

      }

    }


    function checkAuth() {

      if (state.storeName) {

        document.getElementById('authScreen').classList.add('hidden');

        document.getElementById('appScreen').classList.remove('hidden');

        document.getElementById('userName').textContent = state.userName || state.storeName;

        document.getElementById('settingsStoreName').value = state.storeName;

        updateDashboard();

      }

    }


    function handleLogin() {

      const storeName = document.getElementById('storeName').value.trim();

      if (!storeName) {

        showToast('Digite o nome da sua loja', 'error');

        return;

      }

      state.storeName = storeName;

      state.userName = storeName;

      saveState();

      checkAuth();

      showToast('Bem-vindo ao SellHub! 🎉', 'success');

    }


    function logout() {

      if (confirm('Deseja realmente sair?')) {

        localStorage.removeItem('sellhub_state');

        location.reload();

      }

    }


    // ── Navegação ──

    function showSection(sectionId) {

      document.querySelectorAll('.section').forEach(s => s.classList.add('hidden'));

      document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

      document.getElementById(sectionId).classList.remove('hidden');

      event.target.classList.add('active');

      

      if (sectionId === 'dashboard') updateDashboard();

      if (sectionId === 'listings') renderListings();

      if (sectionId === 'stock') renderStock();

      if (sectionId === 'messages') renderMessages();

    }

      document.getElementById('statMessages').textContent = state.messages.length;

        activities.innerHTML = state.listings.slice(0, 5).map(l => `

    // ── Mercado Livre Integration ──

    function connectMercadoLivre() {

      const clientId = prompt('Digite seu Client ID do Mercado Livre:');

      const clientSecret = prompt('Digite seu Client Secret:');

      

      if (!clientId || !clientSecret) {

        showToast('Credenciais necessárias para conectar', 'error');

        return;

      }

      

      // Salvar credenciais (em produção, usar backend seguro)

      state.mlClientId = clientId;

      state.mlClientSecret = clientSecret;

      saveState();

      

      // Redirecionar para OAuth do ML

      const redirectUri = encodeURIComponent(window.location.origin + '/ml-callback');

      const authUrl = `https://auth.mercadolibre.com.br/authorization?response_type=code&client_id=${clientId}&redirect_uri=${redirectUri}&state=${btoa(JSON.stringify({id:clientId,sec:clientSecret}))}`;

      

      window.location.href = authUrl;

    }


    function checkMLCallback() {

      const params = new URLSearchParams(window.location.search);

      const tokenData = params.get('ml_token');

      const error = params.get('ml_error');

      

      if (error) {

        showToast(`Erro ML: ${decodeURIComponent(error)}`, 'error');

        history.replaceState({}, '', window.location.pathname);

        return;

      }

      

      if (tokenData) {

        try {

          const token = JSON.parse(decodeURIComponent(tokenData));

          state.mlToken = token;

          state.mlConnected = true;

          state.userName = token.user_name || state.storeName;

          saveState();

          showToast(`Conectado como ${token.user_name}! 🎉`, 'success');

          updateMLStatus();

          history.replaceState({}, '', window.location.pathname);

        } catch (e) {

          showToast('Erro ao processar token', 'error');

        }

      }

    }


    function updateMLStatus() {

      const status = document.getElementById('mlStatus');

      if (state.mlConnected && state.mlToken) {

        status.textContent = `✅ Conectado (${state.mlToken.user_name})`;

        status.style.color = 'var(--success)';

      } else {

        status.textContent = '❌ Não conectado';

        status.style.color = 'var(--danger)';

      }

    }


    function disconnectMercadoLivre() {

      if (confirm('Desconectar do Mercado Livre?')) {

        state.mlConnected = false;

        state.mlToken = null;

        saveState();

        updateMLStatus();

        showToast('Desconectado', 'success');

      }

    }

      document.getElementById('listingTitle').value = '';

    function saveListing() {

      const title = document.getElementById('listingTitle').value.trim();

      const price = parseFloat(document.getElementById('listingPrice').value);

      const stock = parseInt(document.getElementById('listingStock').value);

      

      if (!title || isNaN(price) || isNaN(stock)) {

        showToast('Preencha todos os campos obrigatórios', 'error');

        return;

      }

      

      const listing = {

        id: Date.now(),

        title,

        price,

        stock,

        category: document.getElementById('listingCategory').value,

        description: document.getElementById('listingDescription').value.trim(),

        createdAt: new Date().toISOString(),

        status: 'active'

      };

      

      state.listings.unshift(listing);

      saveState();

      closeModal('newListingModal');

      renderListings();

      updateDashboard();

      showToast('Anúncio salvo com sucesso! ✅', 'success');

    }


    function editListing(index) {

      const l = state.listings[index];

      document.getElementById('listingTitle').value = l.title;

      document.getElementById('listingPrice').value = l.price;

      document.getElementById('listingStock').value = l.stock;

      document.querySelector('#newListingModal .btn-primary').onclick = () => {

        state.listings.splice(index, 1);

      };

    function renderStock() {

      const container = document.getElementById('stockList');

      

      if (state.stock.length === 0) {

        container.innerHTML = `

          <div class="empty">

            <div class="empty-icon">🏭</div>

        return;

      }

      

      // Agrupar por produto

      const grouped = {};

      state.stock.forEach(s => {

        if (!grouped[s.productId]) grouped[s.productId] = { product: s.product, quantity: 0, movements: [] };

        grouped[s.productId].quantity += s.type === 'entrada' ? s.quantity : -s.quantity;

        grouped[s.productId].movements.push(s);

      });

      

      container.innerHTML = Object.values(grouped).map(g => `

        <div style="padding:16px; border:1px solid var(--border); border-radius:8px; margin-bottom:12px;">

          <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:8px;">

            <strong>${g.product}</strong>

            <span class="badge ${g.quantity > 10 ? 'badge-success' : g.quantity > 0 ? 'badge-warning' : 'badge-danger'}">

              ${g.quantity} unidades

            </span>

          </div>

          <small style="color:var(--gray);">

            ${g.movements.length} movimento${g.movements.length > 1 ? 's' : ''}

          </small>

        </div>

      `).join('');

      

      // Atualizar select do modal

      const select = document.getElementById('stockProduct');

      select.innerHTML = '<option value="">Selecione...</option>' + 

        state.listings.map(l => `<option value="${l.id}">${l.title}</option>`).join('');

    }

      if (state.listings.length === 0) {

        showToast('Crie um anúncio primeiro', 'error');

    function updateStock() {

      const productId = document.getElementById('stockProduct').value;

      const quantity = parseInt(document.getElementById('stockQuantity').value);

      const type = document.getElementById('stockType').value;

      

      if (!productId || isNaN(quantity) || quantity <= 0) {

        showToast('Preencha os campos corretamente', 'error');

        return;

      }

      

      const product = state.listings.find(l => l.id == productId);

      if (!product) return;

      

      state.stock.push({

        id: Date.now(),

        productId,

        product: product.title,

        quantity,

        type,

        date: new Date().toISOString()

      });

      

      saveState();

      closeModal('stockModal');

      renderStock();

      updateDashboard();

      showToast('Estoque atualizado! ✅', 'success');

    }

    function setupEventListeners() {

      // Fechar modal ao clicar fora

      document.querySelectorAll('.modal').forEach(modal => {

        modal.addEventListener('click', (e) => {

          if (e.target === modal) {

            modal.classList.remove('active');

          }

        });

      });

      

      // Enter em inputs do modal

      document.querySelectorAll('.modal input, .modal textarea').forEach(input => {

        input.addEventListener('keypress', (e) => {

          if (e.key === 'Enter' && !e.shiftKey) {

            e.preventDefault();

            const modal = e.target.closest('.modal');

            const saveBtn = modal.querySelector('.btn-primary');

            if (saveBtn) saveBtn.click();

          }

        });

      });

    }

      URL.revokeObjectURL(url);
